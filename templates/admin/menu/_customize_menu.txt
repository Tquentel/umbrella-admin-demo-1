class MyMenu
{
    public function renderMenu(Menu $menu, array $parameters = []): string
    {
        return $this->twig->render('my_nice_template', [
           'menu' => $menu,
       ]);
    }

    public function renderBreadcrumb(Breadcrumb $breadcrumb, array $parameters = []): string
    {
        return $this->twig->render('my_nice_template', [
           'breadcrumb' => $breadcrumb,
       ]);
    }
}
